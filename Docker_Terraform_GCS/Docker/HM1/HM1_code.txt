Q5:East Harlem North
select "PULocationID","Zone", sum(total_amount) 
from public.green_taxi_data t
left join public.taxi_zone_data z on t."PULocationID" = z."LocationID"
where DATE(lpep_pickup_datetime) = '2025-11-18'
group by "PULocationID","Zone"
order by 3 desc
;

Q6: Upper East Side North
select a.*,
z2."LocationID",
z2."Zone" as drop_zone
from 
(select 
"PULocationID" as pickup_id,
"Zone" as pickup_zone,
"DOLocationID" as drop_id,
sum(tip_amount) 
from public.green_taxi_data t
left join public.taxi_zone_data z on t."PULocationID" = z."LocationID"
where date_part('month', lpep_pickup_datetime) = 11
and "PULocationID" = '74'
group by "PULocationID","Zone","DOLocationID"
order by 4 desc) a
left join public.taxi_zone_data z2 on a.drop_id = z2."LocationID"
order by "sum" desc
;
